import time
import matplotlib.pyplot as plt
import numpy as np

from selenium import webdriver
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.common.keys import Keys

driver = webdriver.Chrome('C:/Users/vargv/Anaconda3/Lib/chromedriver')

def get_historic_index(driver, oldest, newest=None):
    '''Enter dates as a string with format : DD.MM.YYYY'''
    
    driver.get('https://www.ariva.de/dax-index/historische_kurse')
    time.sleep(5) # Let the user actually see something!
    
    search_box_min = driver.find_element_by_name('min_time')
    ActionChains(driver).click(search_box_min).double_click(search_box_min).perform()
    search_box_min.send_keys(Keys.DELETE + oldest)
    
    if newest!=None:
        search_box_max = driver.find_element_by_name('max_time')
        ActionChains(driver).click(search_box_max).double_click(search_box_max).perform()
        search_box_max.send_keys(Keys.DELETE + newest)
        
    search_box_min.submit()
    time.sleep(5) # Let the user actually see something!
    driver.quit()
    
get_historic_index(driver, '1.1.2015', '1.1.2017')
